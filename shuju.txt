判断输入的数值有几位
#include<stdio.h>
#pragma warning(disable:4996)
int main()
{
	
	int x ;
	int n = 0;
	scanf("%d", &x);// 输入x
	n++;    // n加1
	x /= 10;// x/10
	while (x > 0){   //判断x是否大于0.大于则执行大括号里面的循环体。直到x不大于0；
		n++;
		x /= 10;
	}
	printf("%d\n", n);
	return 0;
	
}
阶乘 1^N
#include<stdio.h>
#pragma warning(disable:4996)
int main()
{
	
	int n ;
	scanf("%d", &n);
	int fact = 1;
	int i =1;
	while ( i <= n) {
		fact *= i;
		i++;
	
	}
	printf("%d %d\n", n,fact);
	return 0;
	
}
#include<stdio.h>
#pragma warning(disable:4996)
int main()
{

	int i =1;
	for (i = 1; i <= n; i++){
		fact *= i;
	
	}
